    """–û—Ü–µ–Ω–∏–≤–∞–µ—Ç –±–ª–∏–∑–æ—Å—Ç—å –∫ —Å—Ç–∏–ª—é NotebookLM"""
    score = 0.0
    
    # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å—Ç—Ä—É–∫—Ç—É—Ä–∏—Ä–æ–≤–∞–Ω–Ω–æ—Å—Ç–∏
    if "##" in answer and "###" in answer:
        score += 0.2
    
    # –ü—Ä–æ–≤–µ—Ä–∫–∞ –æ–±—ä—è—Å–Ω–µ–Ω–∏—è —á–∏—Å–ª–µ–Ω–Ω—ã—Ö –¥–∞–Ω–Ω—ã—Ö
    numbers_with_units = len(re.findall(r'\d+\.?\d*\s*[–∞-—è]+', answer))
    score += min(numbers_with_units * 0.1, 0.3)
    
    # –ü—Ä–æ–≤–µ—Ä–∫–∞ —Å–≤—è–∑–Ω–æ—Å—Ç–∏
    linking_words = ['–ø–æ—ç—Ç–æ–º—É', '—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ', '—ç—Ç–æ –æ–∑–Ω–∞—á–∞–µ—Ç', '—á—Ç–æ —É–∫–∞–∑—ã–≤–∞–µ—Ç']
    for word in linking_words:
        if word in answer.lower():
            score += 0.1
    
    return min(score, 1.0)
